def theplot_lr(var='crm', label='label', nbins=30, robust=False, **kwargs):
    print '-'*110
    print '::: %s   __vs__   %s'%(var, label)
    print '::: Robust: %r'%robust

    robjects.globalenv[label] = robjects.FloatVector(train[label])
    robjects.globalenv[var] = robjects.FloatVector(train[var])
    if robust==False:
        lm = robjects.r.lm("%s ~ %s"%(label, var))
    else:
        lm = robjects.r.lmrob("%s ~ %s"%(label, var))
    robjects.globalenv["lm_%s"%var] = lm
    coefs_lm = lm.rx2('coefficients')
    print coefs_lm
    
    #f, axes = plt.subplots(1, 2, sharey=True, sharex=True, figsize=(16,4))
    f, axes = plt.subplots(1, 2, sharex=True, figsize=(16,4))
       
    xmin = train[var].min()
    xmax = train[var].max()
    _ = axes[0].scatter(train[var], 
                train[label],
                color='k',
                facecolors='none'
               )
    _ = axes[0].plot([xmin, xmax], 
             [xmin * coefs_lm[1] + coefs_lm[0], xmax * coefs_lm[1] + coefs_lm[0]], 
             color='r',
             linestyle='-',
             linewidth=2)
    
    bin_means, bin_edges, binnumber = scipy.stats.binned_statistic(train[var], 
                                                                   train[label], 
                                                                   statistic='mean', 
                                                                   bins=nbins, 
                                                                   range=(xmin, xmax))
    bin_sem, bin_edges, binnumber = scipy.stats.binned_statistic(train[var], 
                                                                   train[label], 
                                                                   statistic=lambda x: x.std()/np.sqrt(len(x)), 
                                                                   bins=nbins, 
                                                                   range=(xmin, xmax))
    bin_width = (bin_edges[1] - bin_edges[0])
    bin_centers = bin_edges[1:] - bin_width/2
    
    _ = axes[1].errorbar(bin_centers, 
                 bin_means,
                 bin_sem,
                 fmt='ko',
                 mfc='none' )
    _ = axes[1].plot([xmin, xmax],
                 [xmin * coefs_lm[1] + coefs_lm[0],
                  xmax * coefs_lm[1] + coefs_lm[0]],
                 color='k',
                 linestyle='-',
                 linewidth=1)

    for ax in axes:
        ax.set_title('Linear model')
        #ax.legend(loc='upper right')
        ax.set_xlabel(var)
        ax.set_ylabel(label)
        ax.grid()
    plt.tight_layout()
    plt.show()
    #%R plot(lm)

def plot_errors(model='xgb_cv'):
    islog = True
    _ = plt.hist(train['me_%s'%model], 100, log=islog)
    plt.show()
    print '::: me', train['me_%s'%model].mean()
    
    _ = plt.hist(train['mse_%s'%model], 100, log=islog)
    plt.show()
    print '::: mse', train['mse_%s'%model].mean()
    
    _ = plt.hist(train['mpe_%s'%model], 100, log=islog)
    plt.show()
    print '::: mpe', train['mpe_%s'%model].mean()
    
    _ = plt.hist(train['mape_%s'%model], 100, log=islog)
    plt.show()
    print '::: mape', train['mape_%s'%model].mean()
